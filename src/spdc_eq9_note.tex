\noindent
The paper's master formula for the angle-resolved Type-I SPDC photon flux
(used to compute Fig.~6) is printed, in the appendix, as Eq.~(9). Evaluated
\emph{literally}, that printed equation gives an efficiency
$\eta = 2N_s/N_p \approx \num{1.6e-11}$, which is $\sim 17\times$ below the
paper's own Table~I / Fig.~6 value $\eta \approx \num{2.7e-10}$.
Below we trace the chain of substitutions and isolate \textbf{two independent
typos}: a factor of $4$ in the prefactor constant, and a factor
$2\pi n_s^2 \approx 17$ lost in the angular Jacobian.
The answer key (the paper's own quoted $\eta$ and $2N_s\approx\SI{4.2e7}{s^{-1}}$)
decides which forms are correct.

\section*{The relevant printed equations}

\paragraph{Eq.~(7), main text --- flux per unit frequency:}
\begin{equation}
N_s(\omega_s,\kappa_s) =
\frac{\hbar d_{\mathrm{eff}}^2\,\omega_s\omega_i\omega_p L^2 N_p}
{\mathbf{2\pi^4}\, c^3\varepsilon_0 n_s n_i n_p}\,
\dd\omega_s\,\dd^2\kappa_s
\int \dd^2\xi\,
e^{-\frac12\xi^2}\,
\sinc^2\!\left(\tfrac12 L\,\Delta k_z\right).
\tag{7}
\end{equation}
(The exponent is misprinted with a $+$ sign in the paper; it must be
$-\tfrac12\xi^2$, as the appendix has it.)

\paragraph{Appendix intermediate} (after relabeling $\xi\to\xi_i$ and the
$\phi$ saddle-point integral, which supplies the $\sqrt{2\pi}$):
\begin{equation}
N_s =
\underbrace{\frac{\hbar d_{\mathrm{eff}}^2\,\omega_s\omega_i\omega_p L^2 N_p\,\sqrt{2\pi}}
{\mathbf{8\pi^4}\, c^3\varepsilon_0 n_s n_i n_p}}_{\textstyle A_0}\,
\dd\omega_s\,\underbrace{\kappa_s\,\dd\kappa_s\,\dd\phi}_{=\,\dd^2\kappa_s}\,
\int \dd\xi_i\, e^{-\frac12(\xi_s-\xi_i)^2}\,
\sinc^2\!\left(\tfrac12 L\,\Delta k_z\right).
\label{eq:inter}
\end{equation}

\paragraph{Eq.~(9), as printed} (target: variables $\lambda_s,\theta_s$):
\begin{equation}
N_s(\lambda_s,\theta_s) =
2\pi\sqrt{2\pi}\,
\frac{\hbar d_{\mathrm{eff}}^2\,\omega_p L^2 N_p}
{\varepsilon_0 n_s n_i n_p\,\lambda_s^5\lambda_i}\,
\sin(2\theta_s)\,\dd\lambda_s\,\dd\theta_s
\int \dd\xi_i\, e^{-\frac12(\xi_s-\xi_i)^2}\,
\sinc^2\!\left(\tfrac12 L\,\Delta k_z\right).
\tag{9}
\end{equation}

\section*{Error 1: the factor of 4 ($2\pi^4$ vs $8\pi^4$)}

Going from Eq.~(7) to \eqref{eq:inter} the paper changes the constant
$2\pi^4 \to 8\pi^4$ while describing it as the variable relabeling
$\int\dd^2\xi \to \int\dd^2\xi_i$ with $\xi=(\kappa_s+\kappa_i)W$,
$\xi_i=\kappa_i W$. At fixed $\kappa_s$ this is a pure \emph{shift},
\begin{equation}
\int \dd^2\xi\; f(\xi) \;=\; \int \dd^2\xi_i\; f(\xi_s+\xi_i),
\qquad \text{Jacobian}=1,
\end{equation}
which cannot alter the prefactor. Hence one of the two printed constants is a
misprint. Numerically, $8\pi^4$ reproduces $\eta=\num{2.72e-10}$ (matches the
paper); $2\pi^4$ would give $\num{1.09e-9}$, $4\times$ too large.
\textbf{Conclusion: $8\pi^4$ is correct; the main-text $2\pi^4$ is the typo.}

\section*{Error 2: the factor $2\pi n_s^2\approx 17$ (angular Jacobian)}

We carry \eqref{eq:inter} to $(\lambda_s,\theta_s)$ via three substitutions.

\begin{align}
\textbf{(S1)}\quad
\dd\omega_s &= \frac{2\pi c}{\lambda_s^2}\,\dd\lambda_s
&&\left(\omega_s=\tfrac{2\pi c}{\lambda_s}\right),\\[4pt]
\textbf{(S2)}\quad
\int \dd\phi &= 2\pi
&&(\text{isotropic in azimuth}),\\[4pt]
\textbf{(S3)}\quad
\kappa_s\,\dd\kappa_s
&= \tfrac12 k_s^2\,\sin(2\theta_s)\,\dd\theta_s
= \frac{2\pi^2 n_s^2}{\lambda_s^2}\,\sin(2\theta_s)\,\dd\theta_s
&&\left(\kappa_s=k_s\sin\theta_s,\;
k_s=\tfrac{2\pi n_s}{\lambda_s}\right).
\end{align}
Also $\omega_s\omega_i = 4\pi^2 c^2/(\lambda_s\lambda_i)$ inside $A_0$.
Multiplying $A_0$ by (S1)$\cdot$(S2)$\cdot$(S3):
\begin{equation}
P_{\text{correct}}
= 4\pi^2\sqrt{2\pi}\,
\frac{\hbar d_{\mathrm{eff}}^2\,\omega_p L^2 N_p\, n_s}
{\varepsilon_0 n_i n_p\,\lambda_s^5\lambda_i},
\qquad\text{vs.}\qquad
P_{\text{printed}}
= 2\pi\sqrt{2\pi}\,
\frac{\hbar d_{\mathrm{eff}}^2\,\omega_p L^2 N_p}
{\varepsilon_0 n_s n_i n_p\,\lambda_s^5\lambda_i}.
\end{equation}
Their ratio is
\begin{equation}
\boxed{\;
\frac{P_{\text{correct}}}{P_{\text{printed}}}
= \frac{4\pi^2 n_s}{2\pi/n_s}
= 2\pi n_s^2
= 2\pi(1.660)^2 = 17.3.
\;}
\end{equation}

\paragraph{Where it hides.} Writing the full Jacobian
$J$ for $\dd\omega_s\,\dd^2\kappa_s \to \sin(2\theta_s)\,\dd\lambda_s\,\dd\theta_s$,
\begin{equation}
J_{\text{correct}}=\frac{2\pi c}{\lambda_s^2}\cdot\tfrac12 k_s^2\cdot 2\pi
= \frac{8\pi^4 n_s^2 c}{\lambda_s^4},
\qquad
J_{\text{printed}}=\frac{4\pi^3 c}{\lambda_s^4},
\qquad
\frac{J_{\text{correct}}}{J_{\text{printed}}}=2\pi n_s^2 .
\end{equation}
The printed Eq.~(9) retains the $1/\lambda_s^2$ \emph{scaling} of $k_s^2$
(visible as the $\lambda_s^5$ in the denominator) but drops the
$2\pi n_s^2$ \emph{coefficient} of $\tfrac12 k_s^2$. The $n_s^2$ can only come
from $k_s=2\pi n_s/\lambda_s$; thus the slip is entirely in \textbf{(S3)},
the $\kappa_s\,\dd\kappa_s\to\sin(2\theta_s)\,\dd\theta_s$ step.

\section*{Corrected Eq.~(9)}

\begin{equation}
\boxed{\;
N_s(\lambda_s,\theta_s) =
4\pi^2\sqrt{2\pi}\,
\frac{\hbar d_{\mathrm{eff}}^2\,\omega_p L^2 N_p\, n_s}
{\varepsilon_0\, n_i n_p\,\lambda_s^5\lambda_i}\,
\sin(2\theta_s)\,\dd\lambda_s\,\dd\theta_s
\int \dd\xi_i\, e^{-\frac12(\xi_s-\xi_i)^2}\,
\sinc^2\!\left(\tfrac12 L\,\Delta k_z\right)
\;}
\end{equation}
equivalently $N_s = (2\pi n_s^2)\times[\text{printed Eq.~(9)}]$, with
$\xi_s=k_s\sin\theta_s\,W$, $k_s=2\pi n_s/\lambda_s$, and
$\Delta k_z = k_s\cos\theta_s + \sqrt{k_i^2-(\xi_i/W)^2} - k_p$.
This reproduces $\eta=\num{2.72e-10}$ and $2N_s=\SI{4.44e7}{s^{-1}}$
($\theta_m=\SI{29.12}{\degree}$, $P_p=\SI{80}{mW}$), matching the paper's
Table~I. Implemented in \texttt{spdc\_fig6.py}.
